// Шифр Цезаря — Лаб. 1
function encrypt(text, key) {
  return text.split('')
    .map(c => shift(c, key))
    .join('');
}
 
// Виклик:
encrypt("HELLO", 3);
▶ "KHOOR"
 
encrypt("CRYPTO", 13);
▶ "PELCGB"
